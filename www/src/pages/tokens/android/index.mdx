import { version, homepage } from '../../../../../packages/thumbprint-tokens/package.json';
import {
    formatId,
    formatValue,
} from '../../../../../packages/thumbprint-tokens/src/helpers/android.js';
import classNames from 'classnames';
import { camelCase } from 'lodash';
import { graphql } from 'gatsby';
import PackageTable from 'components/package-table';
import TabNav from 'components/thumbprint-tokens/tab-nav';
import TokenSection from 'components/thumbprint-tokens/token-section';

export const pageQuery = graphql`
    {
        allThumbprintToken(
            filter: { tokens: { elemMatch: { value: { android: { ne: null } } } } }
            sort: { fields: name }
        ) {
            nodes {
                name
                description
                tokens {
                    id
                    type
                    value {
                        android
                        web
                    }
                    group
                }
            }
        }
    }
`;

<TabNav />

<PackageTable
    version={version}
    packageName="com.github.thumbtack:thumbprint"
    sourceDirectory={homepage}
    platform="android"
    importStatement="import com.github.thumbtack.thumbprint.tokens.R"
/>

<div>
    {props.data.allThumbprintToken.nodes.map(section => (
        <TokenSection
            section={section}
            formatId={token => formatId(section, token)}
            formatValue={value => formatValue({ value })}
            key={section.name}
            platform="android"
        />
    ))}
</div>
