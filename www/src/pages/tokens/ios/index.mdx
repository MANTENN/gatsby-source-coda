import { name, version, homepage } from '../../../../../packages/thumbprint-tokens/package.json';
import {
    formatId,
    formatValue,
} from '../../../../../packages/thumbprint-tokens/src/helpers/ios.js';
import classNames from 'classnames';
import { camelCase } from 'lodash';
import { graphql } from 'gatsby';
import PackageTable from 'components/package-table';
import TabNav from 'components/thumbprint-tokens/tab-nav';
import TokenSection from 'components/thumbprint-tokens/token-section';

export const pageQuery = graphql`
    {
        allThumbprintToken(
            sort: { order: ASC, fields: [name] }
            filter: { tokens: { elemMatch: { value: { ios: { ne: null } } } } }
        ) {
            edges {
                node {
                    name
                    description
                    tokens {
                        id
                        type
                        value {
                            web
                            ios
                        }
                        group
                    }
                }
            }
        }
    }
`;

<TabNav />

<PackageTable
    version={version}
    packageName={name}
    sourceDirectory={homepage}
    importStatement="import * as tokens from '@thumbtack/thumbprint-tokens';"
/>

<div>
    {props.data.allThumbprintToken.edges.map(({ node: section }) => {
        // Filter out non-iOS tokens.
        section.tokens = section.tokens.filter(token => token.value['ios']);
        return (
            <TokenSection
                section={section}
                idTransform={id => {
                    // Changes 'Border Radius' to 'BorderRadius'.
                    const enumName = section.name.replace(/\s/g, '');
                    return enumName + '.' + formatId({ id });
                }}
                valueTransform={value => {
                    if (value.ios) {
                        return formatValue({ value })
                            .replace('UIColor = ', '')
                            .replace('CGFloat = ', '');
                    }
                }}
                key={section.name}
            />
        );
    })}
</div>
